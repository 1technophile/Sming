81 enum HttpHeaderFieldName {
82  HTTP_HEADER_UNKNOWN = 0,
83 #define XX(_tag, _str, _comment) HTTP_HEADER_##_tag,
84  HTTP_HEADER_FIELDNAME_MAP(XX)
85 #undef XX
86  HTTP_HEADER_CUSTOM // First custom header tag value
87 };
