241  static int onPuback(MqttClient& client, mqtt_message_t* message)
242  {
243  if(!message) {
244  return 1;
245  }
246 
247  if(client.onDelivery) {
248  uint16_t msgId = 0;
249  if(message->common.type == MQTT_TYPE_PUBACK) {
250  msgId = message->puback.message_id;
251  } else if(message->common.type == MQTT_TYPE_PUBREC) {
252  msgId = message->pubrec.message_id;
253  }
254 
255  if(msgId) {
256  client.onDelivery(msgId, (int)message->common.type);
257  }
258  }
259 
260  return 0;
261  }
262 
264  static int onPublish(MqttClient& client, mqtt_message_t* message)
265  {
266  if(message == nullptr) {
267  return -1;
268  }
269 
270  if(message->common.length > MQTT_PAYLOAD_LENGTH) {
271  return -2;
272  }
273 
274  if(client.subscriptionCallback) {
275  String topic = String((const char*)message->publish.topic_name.data, message->publish.topic_name.length);
276  String content;
277  if(message->publish.content.data) {
278  content.concat((const char*)message->publish.content.data, message->publish.content.length);
279  }
280  client.subscriptionCallback(topic, content);
281  }
282 
283  return 0;
284  }
285 #endif
