90 /*
91  * If this is defined, gdbstub will break the program when you press Ctrl-C in gdb.
92  * It does this by monitoring for the 'x03' character in the serial receive routine. Any preceding
93  * characters are passed through to the application via UART2.
94  * If your application uses the serial port for terminal (text) communications you should be OK,
95  * but binary transfers are likely to cause problems and this option should probably be disabled.
96  * Instead, use GDBSTUB_BREAK_ON_INIT, or call gdb_do_break() in your application.
97  *
98  * Specify:
99  * 0 to disable Ctrl+C break checking completely
100  * 1 to allow Ctrl+C break only when debugger is attached
101  * 2 to allow Ctrl+C break at any time. Ensure you have
102  * set remote interrupt-on-connect on
103  * in GDB command file, so it will send a Ctrl+C sequence when attempting to connect
104  */
105 #ifndef GDBSTUB_CTRLC_BREAK
106 #define GDBSTUB_CTRLC_BREAK 2
107 #endif
108 
109 /*
110  * The GDB stub has exclusive access to UART0, so applications cannot use it directly and attempts to
111  * open it will fail.
112  *
113  * If this option is enabled, the default serial port will be changed to UART2 to allow debug output
114  * (from m_printf, debug_*, os_printf, etc.) to show up in your GDB session.
115  *
116  * Outside of GDB terminal applications should work as normal, with the following caveats:
117  *
118  * If GDBSTUB_BREAK_ON_INIT is defined, then at startup your application will display `$T05#b9` and stop.
119  * A similar thing will happen if GDBSTUB_CTRLC_BREAK=2 and you type Ctrl+C.
120  * Continue by typing `$D#44` (without the quotes), or exit the terminal and start GDB.
121  *
122  * See GDB remote serial protocol for further details.
123  *
124  * Disabling this option releases some IRAM. You may be instead able to use UART1 for debug output,
125  * adding `Serial.setPort(UART_ID_1);` in your application's `init()` function.
126 */
127 #ifndef GDBSTUB_ENABLE_UART2
128 #define GDBSTUB_ENABLE_UART2 1
129 #endif
