80 template <uint16_t BUFSIZE> char LineBuffer<BUFSIZE>::addChar(char c)
81 {
82  if(c == '\n' || c == '\r') {
83  return '\n';
84  }
85 
86  if(c >= 0x20 && c < 0x7f && length < (BUFSIZE - 1)) {
87  buffer[length++] = c;
88  buffer[length] = '\0';
89  return c;
90  }

92  return '\0';
