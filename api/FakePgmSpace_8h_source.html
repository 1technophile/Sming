70 // flash memory must be read using 32 bit aligned addresses else a processor exception will be triggered
71 // order within the 32 bit values are
72 // --------------
73 // b3, b2, b1, b0
74 // w1, w0

76 #define pgm_read_with_offset(addr, res) \
77  asm("extui %0, %1, 0, 2\n" /* Extract offset within word (in bytes) */ \
78  "sub %1, %1, %0\n" /* Subtract offset from addr, yielding an aligned address */ \
79  "l32i.n %1, %1, 0x0\n" /* Load word from aligned address */ \
80  "slli %0, %0, 3\n" /* Mulitiply offset by 8, yielding an offset in bits */ \
81  "ssr %0\n" /* Prepare to shift by offset (in bits) */ \
82  "srl %0, %1\n" /* Shift right; now the requested byte is the first one */ \
83  : "=r"(res), "=r"(addr) \
84  : "1"(addr) \
85  :);

207 /*
208  * Create a local (stack) buffer called _name and load it with flash data.
209  * _flash_str is defined locally so the compiler knows its size (length + nul).
210  * Size is rounded up to multiple of 4 bytes for fast copy.
211  *
212  * If defining a string within a function or other local context, must declare static.
213  *
214  * Example:
215  * void testfunc() {
216  * static DEFINE_PSTR(test, "This is a test string\n")
217  * m_printf(LOAD_PSTR(test));
218  * }
219  *
220  */
221 #define LOAD_PSTR(_name, _flash_str) \
222  char _name[ALIGNUP(sizeof(_flash_str))] __attribute__((aligned(4))); \
223  memcpy_aligned(_name, _flash_str, sizeof(_name));
