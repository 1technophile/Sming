39 /*
40  Kelvinator AC map
41 
42  (header mark and space)
43  byte 0 = Basic Modes
44  b2-0 = Modes
45  Modes:
46  000 = Auto (temp = 25C)
47  001 = Cool
48  010 = Dry (temp = 25C, but not shown)
49  011 = Fan
50  100 = Heat
51  b3 = Power Status (1 = On, 0 = Off)
52  b5-4 = Fan (Basic modes)
53  Fan:
54  00 = Auto
55  01 = Fan 1
56  10 = Fan 2
57  11 = Fan 3 or higher (See byte 14)
58  b6 = Vent swing (1 = On, 0 = Off) (See byte 4)
59  b7 = Sleep Modes 1 & 3 (1 = On, 0 = Off)
60  byte 1 = Temperature
61  b3-0: Degrees C.
62  0000 (0) = 16C
63  0001 (1) = 17C
64  0010 (2) = 18C
65  ...
66  1101 (13) = 29C
67  1110 (14) = 30C
68  byte 2 = Extras
69  b3-0 = UNKNOWN, typically 0.
70  b4 = Turbo Fan (1 = On, 0 = Off)
71  b5 = Light (Display) (1 = On, 0 = Off)
72  b6 = Ion Filter (1 = On, 0 = Off)
73  b7 = X-Fan (Fan runs for a while after power off) (1 = On, 0 = Off)
74  byte 3 = Section Indicator
75  b3-0 = Unused (Typically 0)
76  b5-4 = Unknown (possibly timer related) (Typically 0b01)
77  b7-6 = End of command block (B01)
78  (B010 marker and a gap of 20ms)
79  byte 4 = Extended options
80  b0 = Swing Vent Vertical (1 = On, 0 = Off)
81  b4 = Swing Vent Horizontal (1 = On, 0 = Off)
82  byte 5-6 = Timer related. Typically 0 except when timer in use.
83  byte 7 = checksum
84  b3-0 = Unknown (Used in Timer mode)
85  b7-4 = checksum of the previous bytes (0-6)
86  (gap of 40ms)
87  (header mark and space)
88  byte 8 = Repeat of byte 0
89  byte 9 = Repeat of byte 1
90  byte 10 = Repeat of byte 2
91  byte 11 = Section Indicator
92  b3-0 = Unused (Typically 0)
93  b5-4 = Unknown (possibly timer related) (Typically 0b11)
94  b7-6 = End of command block (B01)
95  (B010 marker and a gap of 20ms)
96  byte 12 = Extended options
97  b0 = Sleep mode 2 (1 = On, 0=Off)
98  b6-1 = Unknown (Used in Sleep Mode 3, Typically 0b000000)
99  b7 = Quiet Mode (1 = On, 0=Off)
100  byte 13 = Unknown (Sleep Mode 3 related, Typically 0x00)
101  byte 14 = Fan control
102  b3-0 = Unknown (Sleep Mode 3 related, Typically 0b0000)
103  b6-4 = Fan speed
104  0b000 (0) = Automatic
105  0b001 (1) = Fan 1
106  0b010 (2) = Fan 2
107  0b011 (3) = Fan 3
108  0b100 (4) = Fan 4
109  0b101 (5) = Fan 5
110  byte 15 = checksum
111  b3-0 = Unknown (Typically 0b0000)
112  b7-4 = checksum of the previous bytes (8-14)
113 */
