130 /* Check for UserFrameSize small enough not to require rounding...: */
131  /* Skip 16-byte save area, then 32-byte space for 8 regs of call12
132  * (which overlaps with 16-byte GCC nested func chaining area),
133  * then exception stack frame: */
134  .set UserFrameTotalSize, 16+32+UserFrameSize
135  /* Greater than 112 bytes? (max range of ADDI, both signs, when aligned to 16 bytes): */
136  .ifgt UserFrameTotalSize-112
137  /* Round up to 256-byte multiple to accelerate immediate adds: */
138  .set UserFrameTotalSize, ((UserFrameTotalSize+255) & 0xFFFFFF00)
139  .endif
140 # define ESF_TOTALSIZE UserFrameTotalSize
