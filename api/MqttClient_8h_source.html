237  static int onPuback(MqttClient& client, mqtt_message_t* message)
238  {
239  if(!message) {
240  return 1;
241  }
242 
243  if(client.onDelivery) {
244  uint16_t msgId = 0;
245  if(message->common.type == MQTT_TYPE_PUBACK) {
246  msgId = message->puback.message_id;
247  } else if(message->common.type == MQTT_TYPE_PUBREC) {
248  msgId = message->pubrec.message_id;
249  }
250 
251  if(msgId) {
252  client.onDelivery(msgId, (int)message->common.type);
253  }
254  }
255 
256  return 0;
257  }
258 
259  /* @deprecated This method is only for compatibility with the previous release and will be removed soon. */
260  static int onPublish(MqttClient& client, mqtt_message_t* message)
261  {
262  if(!message) {
263  return -1;
264  }
265 
266  if(message->common.length > MQTT_PAYLOAD_LENGTH) {
267  return -2;
268  }
269 
270  if(client.subscriptionCallback) {
271  String topic = String((const char*)message->publish.topic_name.data, message->publish.topic_name.length);
272  String content;
273  if(message->publish.content.data) {
274  content.concat((const char*)message->publish.content.data, message->publish.content.length);
275  }
276  client.subscriptionCallback(topic, content);
277  }
278 
279  return 0;
280  }
281 #endif
